            \end{itemize}

    \begin{enumerate}

\section{Datensatz}

\begin{frame}{Dataset}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Intel Image Classification:}
            \begin{itemize}
                \item $\sim$25.000 Bilder
                \item 150×150 RGB
                \item 6 Klassen (balanciert)
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Split:}
            \begin{itemize}
                \item Train: $\sim$14.000
                \item Validation: $\sim$3.000
                \item Test: $\sim$7.000
            \end{itemize}
        \end{column}
        
        \begin{column}{0.45\textwidth}
            \textbf{Preprocessing:}
            \begin{itemize}
                \item Normalisierung [0,1]
                \item Data Augmentation:
                \begin{itemize}
                    \item Rotation (±20°)
                    \item Horizontal Flip
                    \item Zoom (10\%)
                    \item Shift (10\%)
                \end{itemize}
                \item Batch Size: 32
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% SECTION 3: Modellarchitekturen

\section{Modellarchitekturen}

\begin{frame}{Modellübersicht}
    \begin{center}
        \begin{tabular}{lccc}
            \toprule
            \textbf{Modell} & \textbf{Parameter} & \textbf{Val Accuracy} & \textbf{Strategie} \\
            \midrule
            Baseline CNN & $\sim$2M & 73\% & Simple CNN \\
            Regularized CNN & $\sim$2M & 79\% & + L2 + Dropout \\
            MobileNetV2 & $\sim$3.5M & 85-90\% & Transfer Learning \\
            \bottomrule
        \end{tabular}
    \end{center}
    
    \vspace{0.3cm}
    
    \textbf{Implementierung:}
    \begin{itemize}
        \item \texttt{src/models/}: baseline.py, regularized.py, transfer\_learning.py
        \item \texttt{src/training/}: Unified trainer + callbacks
        \item \texttt{scripts/train\_pipeline.py}: Komplette Pipeline
    \end{itemize}
\end{frame}

\begin{frame}{Modelle im Detail}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Baseline CNN:}
            \begin{itemize}
                \item 3× Conv2D + MaxPooling
                \item Flatten + Dense
                \item Dropout (0.5)
                \item Problem: Overfitting (73\% val)
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Regularized CNN:}
            \begin{itemize}
                \item + Batch Normalization
                \item + L2 Regularization
                \item + Dropout (0.6)
                \item Besser: 79\% val
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{MobileNetV2:}
            \begin{itemize}
                \item Pre-trained (ImageNet)
                \item Custom Head:
                \begin{itemize}
                    \item GlobalAvgPooling
                    \item Dense(256)
                    \item Dropout(0.5)
                    \item Dense(6)
                \end{itemize}
                \item Beste: 85-90\% val
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\section{Lessons Learned}

\begin{frame}{Lessons Learned}
    \textbf{Technische Erkenntnisse:}
    
    \begin{itemize}
        \item \textbf{Transfer Learning:} MobileNetV2 deutlich besser als Custom CNNs
        \item \textbf{Regularisierung:} Batch Norm + L2 + Dropout reduziert Overfitting
        \item \textbf{Data Augmentation:} Essentiell für Generalisierung
        \item \textbf{FastAPI:} Schnelle API-Entwicklung mit Auto-Docs
        \item \textbf{Streamlit:} Einfache Dashboard-Erstellung
        \item \textbf{Docker Compose:} Vereinfacht Multi-Service Deployment
    \end{itemize}
\end{frame}
